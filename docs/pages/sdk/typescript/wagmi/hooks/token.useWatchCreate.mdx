# `token.useWatchCreate`

Watches for new TIP20 token creation events on the TIP20 Factory.

## Usage

:::code-group

```ts twoslash [example.ts]
// @errors: 2322
import { config } from './wagmi.config'
declare module 'wagmi' {
  interface Register {
    config: typeof config
  }
}
// ---cut---
import { Hooks } from 'tempo.ts/wagmi'

Hooks.token.useWatchCreate({
  onTokenCreated: (args, log) => {
    console.log('Creator:', args.creator)
    console.log('Name:', args.name)
    console.log('Symbol:', args.symbol)
    console.log('Token address:', args.tokenAddress)
    console.log('Token ID:', args.tokenId)
  },
})
```


:::
