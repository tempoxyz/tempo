# `dex.useWatchOrderFilled`

Watches for order filled events on the Stablecoin Exchange.

## Usage

:::code-group

```ts twoslash [example.ts]
// @errors: 2322
import { config } from './wagmi.config'
declare module 'wagmi' {
  interface Register {
    config: typeof config
  }
}
// ---cut---
import { Hooks } from 'tempo.ts/wagmi'

Hooks.dex.useWatchOrderFilled({
  onOrderFilled: (args, log) => {
    console.log('Order filled:', args.orderId)
    console.log('Maker:', args.maker)
    console.log('Taker:', args.taker)
    console.log('Amount:', args.amount)
  },
})
```
