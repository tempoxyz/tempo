# `token.useWatchBurn`

Watches for token burn events on TIP20 tokens.

## Usage

:::code-group

```ts twoslash [example.ts]
// @errors: 2322
import { config } from './wagmi.config'
declare module 'wagmi' {
  interface Register {
    config: typeof config
  }
}
// ---cut---
import { Hooks } from 'tempo.ts/wagmi'

Hooks.token.useWatchBurn({
  onBurn: (args, log) => {
    console.log('Amount:', args.amount)
    console.log('Burner:', args.burner)
    console.log('From:', args.from)
  },
  token: '0x20c0000000000000000000000000000000000000',
})
```

```ts twoslash [wagmi.config.ts] filename="wagmi.config.ts"
// @noErrors
// [!include ~/snippets/wagmi.config.ts:setup]
```

:::
