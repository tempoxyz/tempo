# `dex.useWatchOrderPlaced`

Watches for order placed events on the Stablecoin Exchange.

## Usage

:::code-group

```ts twoslash [example.ts]
// @errors: 2322
import { config } from './wagmi.config'
declare module 'wagmi' {
  interface Register {
    config: typeof config
  }
}
// ---cut---
import { Hooks } from 'tempo.ts/wagmi'

Hooks.dex.useWatchOrderPlaced({
  onOrderPlaced: (args, log) => {
    console.log('Order placed:', args.orderId)
    console.log('Maker:', args.maker)
    console.log('Token:', args.token)
    console.log('Amount:', args.amount)
    console.log('Tick:', args.tick)
  },
})
```


:::
